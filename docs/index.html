VueJackBox

VueJackBox is a plugin to display beautiful dialogs in vue. GitHub.

Dialog state

This affects the styling on all dialogs in this demo.

Builder

{{ title }}
{{ message }}
{{ question }}
this.$dialog({{ code }})

Confirm 

  this.$confirm({
    title: 'Confirm',
    message: 'This is a confirm',
    state: '{{ state }}',
  });

Alert 

    this.$prompt({
      title: 'Prompt',
      message: 'This is an prompt',
      state: '{{ state }}',
      ok: {
        text: 'Confirm',
        action: this.callback,
      },
      value: this.val
    });

Prompt 

  this.$prompt({
    title: 'Prompt',
    message: 'This is an prompt',
    state: '{{ state }}',
    ok: {
      text: 'Confirm',
      action: (val) => { this.val = val; }),
    },
    value: this.val
  });

Notification 

    this.$notification({
      title: 'Notification',
      message: 'This is an notification',
      state: '{{ state }}',
    });
